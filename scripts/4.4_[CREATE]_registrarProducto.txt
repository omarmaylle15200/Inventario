create PROCEDURE sp_registrarProducto(
                                      in v_codigo varchar(50),
    								  in v_nombre varchar(50),
                                      in v_descripcion text,
                                      in v_precioCompra decimal(10,2),
                                      in v_precioVenta decimal(10,2),
    								  in v_cantidad int,
    								  in v_idCategoria int,
                                      in v_idProveedor int,
                                      in v_idUsuarioRegistro int)
BEGIN
	insert into producto(Codigo,Nombre,Descripcion,PrecioCompra,PrecioVenta,Cantidad,IdCategoria,IdProveedor,IdUsuarioRegistro)
    values (v_codigo,v_nombre,v_descripcion,v_precioCompra,v_precioVenta,v_cantidad,v_idCategoria,v_idProveedor,v_idUsuarioRegistro);
END //
