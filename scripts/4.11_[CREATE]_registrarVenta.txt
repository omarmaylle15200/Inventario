DELIMITER //
create PROCEDURE sp_registrarVenta(in v_idTipoDocumentoVenta int,
                                      in v_idCliente varchar(50),
                                      in v_total decimal(10,2),
                                      in v_idUsuarioRegistro int,
                                  	  out v_idVenta int)
BEGIN
	insert into venta (IdTipoDocumentoVenta,IdCliente,Total,IdUsuarioRegistro)
    values (v_idTipoDocumentoVenta,v_idCliente,v_total,v_idUsuarioRegistro);
    
    set v_idVenta=(SELECT max(IdVenta) from venta);
    select v_idVenta as IdVenta;
END //
