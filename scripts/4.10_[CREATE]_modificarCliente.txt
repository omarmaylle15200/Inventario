DELIMITER //
create PROCEDURE sp_modificarCliente(in v_idCliente int,
                                      in v_numeroDocumento varchar(20),
    								  in v_nombre varchar(100),
                                      in v_direccion text,
                                      in v_email varchar(100),
                                      in v_telefono varchar(20),
						  			  in v_esActivo bit,
                                      in v_idUsuarioModificacion int)
BEGIN
	
    update cliente
    SET
    	NumeroDocumento=v_numeroDocumento,
    	Nombre=v_nombre,
        Direccion=v_direccion,
        Email=v_email,
        Telefono=v_telefono,
        EsActivo=v_esActivo,
        FechaModificacion=now(),
        IdUsuarioModificacion=v_idUsuarioModificacion
    WHERE
    IdCliente=v_idCliente;
END //
