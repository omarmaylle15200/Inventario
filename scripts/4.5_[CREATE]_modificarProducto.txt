DELIMITER //
create PROCEDURE sp_modificarProducto(in v_idProducto int,
                                      in v_codigoReferencia varchar(50),
                                      in v_modelo varchar(50),
                                      in v_nombre varchar(50),
                                      in v_descripcion text,
                                      in v_cantidad int,
                                      in v_idCategoria int,
                                      in v_idProveedor int,
                                      in v_precioCompra decimal(10,2),
                                      in v_precioVenta decimal(10,2),
                                      in v_idUsuarioModificacion int)
BEGIN
	
    update producto
    SET
    	Nombre=v_nombre,
        Descripcion=v_descripcion,
        Modelo=v_modelo,
        CodigoReferencia=v_codigoReferencia,
        Cantidad=v_cantidad,
        IdCategoria=v_idCategoria,
        IdProveedor=v_idProveedor,
        PrecioCompra=v_precioCompra,
        PrecioVenta=v_precioVenta,
        FechaModificacion=now(),
        IdUsuarioModificacion=v_idUsuarioModificacion
    WHERE
    IdProducto=v_idProducto;
END //
