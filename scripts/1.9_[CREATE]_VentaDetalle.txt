CREATE TABLE VentaDetalle (
    IdVentaDetalle BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    IdVenta BIGINT,
    Item int,
    IdProducto INT,
    Cantidad int,
    Precio decimal(10,2),
    SubTotal decimal(10,2)
);
