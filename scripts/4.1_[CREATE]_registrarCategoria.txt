create PROCEDURE sp_registrarCategoria(in v_nombre varchar(50),
                                       	in v_descripcion varchar(100),
                                       in v_idUsuarioRegistro int)
BEGIN
	set @codigo=(SELECT concat('C',RIGHT(concat('00000',(SELECT COUNT(IdCategoria)+1 from categoria)),5)));
    
	insert into categoria (Codigo,Nombre,Descripcion,IdUsuarioRegistro)
    values (@codigo,v_nombre,v_descripcion,v_idUsuarioRegistro);
END //
