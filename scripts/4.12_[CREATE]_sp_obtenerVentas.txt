create PROCEDURE sp_obtenerVentas()
BEGIN
	
    SELECT 
        v.IdVenta,
        v.IdTipoDocumentoVenta,
        tdv.Descripcion tipoDocumentoVenta,
        v.IdCliente,
        c.NumeroDocumento numeroDocumentoCliente,
        c.Nombre nombreCliente,
        v.Total,
        v.FechaRegistro
    FROM venta v 
    inner join tipodocumentoventa tdv on tdv.IdTipoDocumentoVenta=v.IdTipoDocumentoVenta
    inner join cliente c on c.IdCliente=v.IdCliente;
    
END //
