DELIMITER //
create PROCEDURE sp_registrarProveedor(
    								  in v_idTipoDocumento int,
    								  in v_numeroDocumento varchar(20),
    								  in v_nombre varchar(100),
                                      in v_direccion text,
                                      in v_email varchar(100),
                                      in v_telefono varchar(20),
                                      in v_idUsuarioRegistro int)
BEGIN
	insert into proveedor (NumeroDocumento,Nombre,Direccion,Email,Telefono,IdUsuarioRegistro)
    values (v_numeroDocumento,v_nombre,v_direccion,v_email,v_telefono,v_idUsuarioRegistro);
END
